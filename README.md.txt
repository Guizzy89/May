MyStore/
  ├── DataAccessLayer/      # слой доступа к данным (EF Core + репозитории)
  │   ├── Models/           # классы моделей данных
  │   └── DbContext.cs      # контекст базы данных
  ├── BusinessLogicLayer/   # слой бизнес-логики (сервисы и менеджеры)
  │   ├── Services/         # сервисы для операций над моделями
  │   └── Managers/         # менеджерские классы
  ├── WebApi/               # точка входа API
  │   ├── Controllers/      # контроллеры RESTful API
  │   ├── Filters/          # фильтры безопасности
  │   └── Startup.cs        # настройка конфигурации сервера
  ├── Program.cs           # стартовая точка программы
  └── Tests/                # тесты для покрытия функциональности
